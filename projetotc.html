// C++ code
//
void setup()
{
pinMode(8, OUTPUT);
pinMode(13, OUTPUT);
pinMode(9, OUTPUT);
pinMode(10, OUTPUT);
pinMode(11, OUTPUT);
pinMode(12, OUTPUT);
}

void loop()
{
digitalWrite(8, HIGH);
digitalWrite(13, HIGH);
delay(9000); // Wait for 9000 millisecond(s)
delay(9000); // Wait for 9000 millisecond(s)
digitalWrite(8, LOW);
digitalWrite(9, HIGH);
digitalWrite(13, LOW);
delay(2000); // Wait for 2000 millisecond(s)
digitalWrite(9, LOW);
digitalWrite(10, HIGH);
digitalWrite(11, HIGH);
delay(7000); // Wait for 7000 millisecond(s)
delay(7000); // Wait for 7000 millisecond(s)
digitalWrite(10, LOW);
digitalWrite(12, HIGH);
digitalWrite(11, LOW);
delay(2000); // Wait for 2000 millisecond(s)
digitalWrite(12, LOW);
}
